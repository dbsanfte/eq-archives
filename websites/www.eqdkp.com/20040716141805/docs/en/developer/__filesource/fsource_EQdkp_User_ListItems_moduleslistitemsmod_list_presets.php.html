50    /**
51     * List Presets module for ListItems
52     *
53     * @author Rob Speicher <tsigo@eqdkp.com>
54     * @package EQdkp_User
55     * @subpackage ListItems
56     * @version $Id: mod_list_presets.php,v 1.1 2003/12/24 00:35:16 tsigo Exp $

59    class Listitems_List_presets
60    {
61        /**
62         * Reference to base class object that loaded this module
63         * @access private
64         * @var object ref 
65         */
66        var $loader = null;
67    
68        /**
69         * Location of this module in the Module_Base::modules array
70         * @access private
71         * @var int 
72         */
73        var $index = -1;
74    
75        /**
76         * Number of items listed
77         * @access private
78         * @var int 
79         */
80        var $num_items = 0;
81    
82        /**
83         * Pagination start variable
84         * @access private
85         * @var string 
86         */
87        var $start_var = 'start';
88    
89        /**
90         * Pagination start number
91         * @access private
92         * @var int 
93         */
94        var $start = 0;
95    
96        /**
97         * @access private
98         * @var array 
99         */
100       var $sort_order = array();
101   
102       /**
103        * Set the values of {@link $loader} and {@link $index},
104        * add the module's language pack and tell Module_Base this module's
105        * initialization methods.
106        *
107        * @param object ref Setting for {@link $loader}
108        * @param int        Setting for {@link $index}
109        */
110       function listitems_list_presets(&$module_loader, $module_index)
111       {
112           global $eqdkp $user $tpl
113   
114           $this->loader = $module_loader;
115           $this->index  = $module_index;
116   
117           $user->add_lang('modules/listitems/', 'list_presets');
118   
119           if ( $eqdkp->input['type'] == ITM_PRESETS )
120           {
121               $this->loader->add_startmethod($this->index, array(
122                   'header',
123                   'preset_rows',
124                   'footer')
125               );
126   
127               $tpl->set_vars(array(
128                   'page_title' => $eqdkp->title($user->lang['TITLE_PRESETS']))
129               );
130           }
131       }
132   
133       /**
134        * Output basic header values for this module and set up sorting
135        *
136        * @access protected
137        */
138       function header()
139       {
140           global $eqdkp $tpl
141   
142           //--------------------------------
143           // Handle sorting
144           //--------------------------------
145   
146           $sort_order = array(
147               0 => array('preset_name', 'preset_name desc'),
148               1 => array('preset_value desc', 'preset_value')
149           );
150   
151           $this->sort_order = $eqdkp->sort_order($sort_order);
152   
153           //--------------------------------
154           // Assign sort parameters to template
155           //--------------------------------
156   
157           $tpl->assign(array(
158               'O_ITEM'  => $this->sort_order[0],
159               'O_VALUE' => $this->sort_order[1])
160           );
161       }
162   
163       /**
164        * Output item preset rows
165        *
166        * @access protected
167        */
168       function preset_rows()
169       {
170           global $eqdkp $user $db $tpl
171           global $SID
172   
173           $this->start = ( isset($eqdkp->input[$this->start_var]) ) ? intval($eqdkp->input[$this->start_var]) : 0;
174   
175           $sql = 'SELECT preset_id, preset_name, preset_value
176                   FROM ' . ITEM_PRESETS_TABLE . '
177                   ORDER BY ' . $this->sort_order['SQL'] . '
178                   LIMIT ' . $this->start . ',' . $user->data['user_ilimit'];
179           $result = $db->query($sql);
180           while ( $row = $db->fetch_record($result) )
181           {
182               $tpl->append_block('ITEMS_ROW', array(
183                   'U_VIEWITEM'  => 'viewitem.php' . $SID . '&amp;' . URI_ITEM . '=' . $row['preset_id'],
184                   'U_EDITITEM'  => 'admin/additem.php' . $SID . '&amp;' . URI_ITEM . '=' . $row['preset_id'] . '&amp;type=' . ITM_PRESET,
185                   'ID'          => $row['preset_id'],
186                   'NAME'        => ( $row['preset_name'] != '' ) ? stripslashes($row['preset_name']) : '&nbsp;',
187                   'VALUE'       => $row['preset_value'])
188               );
189           }
190           $db->free_result($result);
191   
192           $tpl->assign_block('ITEMS_ROW');
193       }
194   
195       /**
196        * Output raid pagination
197        *
198        * @access protected
199        */
200       function footer()
201       {
202           global $eqdkp $user $db $tpl
203   
204           $sql = 'SELECT count(preset_id)
205                   FROM ' . ITEM_PRESETS_TABLE;
206           $this->num_items = $db->query_first($sql);
207   
208           $tpl->assign(array(
209               'USER_ILIMIT'     => $user->data['user_ilimit'],
210               'NUM_ITEMS'       => $this->num_items,
211               'ITEM_PAGINATION' => $eqdkp->pagination($this->num_items, $user->data['user_ilimit'], $this->start, $this->start_var))
212           );
213       }
214   }
