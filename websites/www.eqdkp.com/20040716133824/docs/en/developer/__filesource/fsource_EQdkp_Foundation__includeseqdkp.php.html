85        /**
86         * Get data from CONFIG_TABLE and store it in the {@link $config} array
87         *
88         * @access public
89         * @return bool 
90         */
91        function config_init()
92        {
93            global $db
94    
95            if ( !is_object($db) )
96            {
97                trigger_error('Database object not instantiated, cannot get EQdkp configuration data!', E_USER_ERROR);
98                return false;
99            }
100   
101           $sql = 'SELECT config_name, config_value
102                   FROM ' . CONFIG_TABLE;
103           $result = $db->query($sql);
104           while ( $row = $db->fetch_record($result) )
105           {
106               $this->config[ $row['config_name'] ] = $row['config_value'];
107           }
108   
109           return true;
110       }

112       /**
113        * Set a config value in CONFIG_TABLE
114        *
115        * Saves developers from running an SQL query to update a config value
116        *
117        * @access public
118        * @param string|array   config_name field value; if array, recurse
119        * @param string          config_value field value
120        * @return bool 
121        */
122       function config_set($config_name, $config_value = '')
123       {
124           global $db
125   
126           if ( is_object($db) )
127           {
128               if ( is_array($config_name) )
129               {
130                   // Recurse through the $config_name array
131                   foreach ( $config_name as $d_name => $d_val )
132                   {
133                       $this->config_set($d_name, $d_val);
134                   }
135               }
136               else
137               {
138                   $sql = 'UPDATE ' . CONFIG_TABLE . "
139                           SET config_value = '" . $config_value . "'
140                           WHERE config_name = '" . $config_name . "'";
141                   $db->query($sql);
142   
143                   return true;
144               }
145           }
146           else
147           {
148               return false;
149           }
150       }

152       /**
153        * Clean $_GET and $_POST variables and store them in {@link $input} array
154        *
155        * $_GET vars are stored first, so any duplicates in $_POST overwrite them.
156        *
157        * @access public
158        */
159       function parse_incoming_data()
160       {
161           global $REQUEST_METHOD
162   
163           //--------------------------------
164           // Clean _GET vars
165           //--------------------------------
166   
167           if ( is_array($_GET) )
168           {
169               foreach ( $_GET as $key => $val )
170               {
171                   $this->input[ $this->clean_key($key) ] = $this->clean_data($val);
172               }
173           }
174   
175           //--------------------------------
176           // Clean _POST vars and overwrite _GET vars
177           //--------------------------------
178   
179           if ( is_array($_POST) )
180           {
181               foreach ( $_POST as $key => $val )
182               {
183                   $this->input[ $this->clean_key($key) ] = $this->clean_data($val);
184               }
185           }
186   
187           //--------------------------------
188           // Figure out the request method
189           //--------------------------------
190   
191           $this->input['REQUEST_METHOD'] = ( $_SERVER['REQUEST_METHOD'] != '' ) ? strtolower($_SERVER['REQUEST_METHOD']) : strtolower($REQUEST_METHOD);
192       }

194       /**
195        * Cleans input values; recurses for values that are arrays
196        *
197        * @access private
198        * @param mixed Data to clean
199        * @return mixed Cleaned data
200        */
201       function clean_data($data)
202       {
203           if ( is_array($data) )
204           {
205               foreach ( $data as $key => $val )
206               {
207                   if ( is_array($val) )
208                   {
209                       $data[$key] = $this->clean_data($val);
210                   }
211                   else
212                   {
213                       unset($data[$key]);
214   
215                       $key = $this->clean_key($key);
216                       $val = $this->clean_val($val);
217   
218                       $data[$key] = $val;
219                   }
220               }
221           }
222           else
223           {
224               $data = $this->clean_val($data);
225           }
226   
227           return $data;
228       }

247       /**
248        * Clean an input value for data that could interfere with SQL queries
249        * or with page display.
250        *
251        * @access private
252        * @param string Value to clean
253        * @return string Cleaned value
254        */
255       function clean_val($val)
256       {
257           if ( $val == '' )
258           {
259               return '';
260           }
261   
262           $val = str_replace('&#032;', ' ',            $val);            // 32   Non-breaking space
263           $val = str_replace('&',      '&amp;',        $val);            // XHTML-friendly ampersand escape
264           $val = str_replace('$',      '&#36;',        $val);
265           $val = str_replace('!',      '&#33;',        $val);
266   
267           $val = str_replace('>',      '&gt;',         $val);
268           $val = str_replace('<',      '&lt;',         $val);
269           $val = str_replace('"',      '&quot;',       $val);            // Convert double quotes
270           $val = str_replace("'",      '&#39;',        $val);            // Convert single quotes
271   
272           $val = str_replace('<!--',   '&#60;&#33;--', $val);            // Get rid of HTML comments, possible SSI
273           $val = str_replace('-->',    '--&#62;',      $val);
274   
275           $val = str_replace("\n",     '<br />',       $val);            // Convert newlines
276           $val = str_replace("\r",     '',             $val);            // Remove carriage returns
277   
278           // Undo Magic Quotes if they're on
279           if ( get_magic_quotes_gpc() )
280           {
281               $val = stripslashes($val);
282           }
283   
284           $val = preg_replace('/\\\(?!&amp;#|\?#)/', '&#92;', $val);     // Get rid of user-input backslashes
285   
286           return $val;
287       }

289       /**
290        * Return a string that contains the name of this page as well as any
291        * requested input var=val pairs.
292        *
293        * Loops through {@link $input} and appends the input variables to a URL
294        * string that begins with the filename of the current page.  If the input
295        * key is found in $skip then that variable isn't appended to the string.
296        * <br /><br />
297        * Example:<br />
298        *      Current page is "listmembers.php?s=&rank=Member&filter=Plate"<br />
299        *      $eqdkp->uri_self(true, 'rank') results in "listmembers.php?s=&amp;filter=Plate"<br />
300        *      $eqdkp->uri_self(true, 's,filter') results in "listmembers.php?rank=Member"<br />
301        *
302        * @access public
303        * @param bool      True: use '&amp;', False: use '&' in query string
304        * @param string    Comma-separated list of input keys to skip
305        */
306       function uri_self($escape = true, $skip = '')
307       {
308           global $SID
309   
310           $amp  = ( $escape ) ? '&amp;' : '&';
311           $skip = array_merge(array(URI_SESSION), explode(',', $skip));
312   
313           $uri_string = '';
314   
315           if ( (isset($this->input['REQUEST_METHOD'])) && ($this->input['REQUEST_METHOD'] == 'get') )
316           {
317               foreach ( $this->input as $key => $val )
318               {
319                   if ( (!is_array($val)) && ($key != 'REQUEST_METHOD') && (!in_array($key, $skip)) && (!in_array('*', $skip)) )
320                   {
321                       $uri_string .= $amp . $key . '=' . $val;
322                   }
323               }
324   
325               $uri_string = $SID . $uri_string;
326           }
327   
328           return $_SERVER['PHP_SELF'] . $uri_string;
329       }

331       /**
332        * Create a random alpha-numeric string suitable for use as a password
333        *
334        * Password will be [A-Za-z0-9]{8}
335        *
336        * @access public
337        * @param bool Return string as an md5 hash (true) or as a plaintext string (false)
338        * @return string 
339        */
340       function random_string($hash = false)
341       {
342           $chars = array('a','A','b','B','c','C','d','D','e','E','f','F','g','G','h','H','i','I','j','J',
343                          'k','K','l','L','m','M','n','N','o','O','p','P','q','Q','r','R','s','S','t','T',
344                          'u','U','v','V','w','W','x','X','y','Y','z','Z','1','2','3','4','5','6','7','8',
345                          '9','0');
346   
347           $max_chars = count($chars) - 1;
348           srand((double)microtime() * 1000000);   // floatval() is only available in PHP >= 4.2.0
349   
350           $rand_str = '';
351           for($i = 0; $i < 8; $i++)
352           {
353               $rand_str = ( $i == 0 ) ? $chars[rand(0, $max_chars)] : $rand_str . $chars[rand(0, $max_chars)];
354           }
355   
356           return ( $hash ) ? md5($rand_str) : $rand_str;
357       }

359       /**
360        * Defines the sorting order of the supplied array
361        *
362        * The supplid array is in the format [index][0 or 1] (0 = the default order, 1 = the opposite).
363        * Returns an array containing the string to use in an SQL query and an
364        * array to use to pass the sort value through the URI.  URI string is in
365        * the format (index).(0/1)
366        *
367        * @access public
368        * @param array Sort Order
369        * @return array 
370        */
371       function sort_order($sort_order)
372       {
373           $uri_order = ( isset($this->input[URI_ORDER]) ) ? $this->input[URI_ORDER] : '0.0';
374           $uri_order = explode('.', $uri_order);
375   
376           $element1 = ( isset($uri_order[0]) ) ? $uri_order[0] : 0;
377           $element2 = ( isset($uri_order[1]) ) ? $uri_order[1] : 0;
378   
379           // Make sure that our indices aren't bigger than the arrays
380           $array_size = sizeof($sort_order);
381           if ( $element1 > $array_size - 1 )
382           {
383               $element1 = $array_size - 1;
384           }
385   
386           if ( $element2 > 1 )
387           {
388               $element2 = 0;
389           }
390   
391           for ( $i = 0; $i < $array_size; $i++ )
392           {
393               if ( $element1 == $i )
394               {
395                   $uri_element2 = ( $element2 == 0 ) ? 1 : 0;
396               }
397               else
398               {
399                   $uri_element2 = 0;
400               }
401   
402               $current_order[$i] = $i . '.' . $uri_element2;
403           }
404   
405           $current_order['CURRENT'] = $element1 . '.' . $element2;
406           $current_order['SQL']     = $sort_order[$element1][$element2];
407   
408           return $current_order;
409       }

411       /**
412        * Perform an SQL-like column sort on an array
413        *
414        * Allows us to sort DKP data stored only in an array.  $array_args must be
415        * an array in the format array('field_name', SORT_ORDER) or a string in the
416        * format 'field_name SORT_ORDER', where SORT_ORDER is SORT_ASC or SORT_DESC.
417        * field_name MUST exist as an array key in each array value in $array.
418        *
419        * @access public
420        * @param array     The DKP data array
421        * @param mixed     Field and order to sort by.
422        * @return array Sorted array
423        */
424       function array_csort($array, $array_args)
425       {
426           if ( !is_array($array_args) )
427           {
428               $array_args = split(' ', $array_args);
429           }
430   
431           foreach ( $array as $row )
432           {
433               $sort_array[] = $row[ $array_args[0] ];
434           }
435   
436           $sort_order = constant($array_args[1]);
437   
438           array_multisort($sort_array, $sort_order, $array);
439   
440           return $array;
441       }

443       /**
444        * Create a string of pagination links
445        *
446        * Returns a "< 1 2 3 4 [5] 6 7 >"-type string.
447        * Define $start_variable if multiple paginations exist on a single page.
448        *
449        * @access public
450        * @param int       Total number of items to paginate
451        * @param int       Number of items per page
452        * @param int       Number to start on, determines what page we're on
453        * @param string    Name of start input variable, defaults to 'start'
454        * @param string    String to append to end of page links
455        * @return string 
456        */
457       function pagination($num_items, $per_page, $start_item, $start_variable = 'start', $append = '')
458       {
459           global $user
460   
461           $base_url    = $this->uri_self(true, $start_variable);
462           $total_pages = ceil($num_items / $per_page);
463           $on_page     = floor($start_item / $per_page) + 1;
464           $initial_i   = 2;
465           $ending_i    = $total_pages;
466   
467           if ( ($total_pages == 1) || (!$num_items) )
468           {
469               return '';
470           }
471   
472           $pagination = '';
473   
474           //--------------------------------
475           // Previous page / Page 1 links
476           //--------------------------------
477   
478           if ( $on_page == 1 )
479           {
480               $pagination .= '<span style="font-weight: bold">1</span>' . "\n";
481           }
482           else
483           {
484               $title = sprintf($user->lang['PAGE_X'], 1);
485               $title .= ' (1 - ' . $per_page . ')';
486   
487               $pagination .= '<a href="' . $base_url . '&amp;' . $start_variable . '=' . (($on_page - 2) * $per_page) . $append .
488                               '" title="' . $user->lang['PAGE_PREVIOUS'] . '" class="pagination">&lt;</a>' . "\n";
489               $pagination .= ' <a href="' . $base_url . '&amp;' . $start_variable . '=0' . $append . '" title="' . $title . '" class="pagination">1</a>' . "\n";
490           }
491   
492           //--------------------------------
493           // Page 2 to page (n - 1) links
494           //--------------------------------
495   
496           if ( $total_pages > 5 )
497           {
498               // Insert prefix ellipses if necessary
499               $start_count = min(max(1, $on_page - 6), $total_pages - 5);
500               $end_count   = max(min($total_pages, $on_page + 6), 5);
501   
502               $pagination .= ( $start_count > 1 ) ? ' ... ' . "\n" : ' ';
503   
504               $initial_i = $start_count + 1;
505               $ending_i  = $end_count;
506           }
507   
508           // Insert individual page links
509           for ( $i = $initial_i; $i < $ending_i; $i++ )
510           {
511               if ( $on_page == $i )
512               {
513                   $pagination .= '<span style="font-weight: bold">' . $i . '</span>' . "\n";
514               }
515               else
516               {
517                   $title = sprintf($user->lang['PAGE_X'], $i);
518                   $title .= ' (' . ((($i - 1) * $per_page) + 1) . ' - ' . ((($i - 1) * $per_page) + $per_page) . ')';
519   
520                   $pagination .= '<a href="' . $base_url . '&amp;' . $start_variable . '=' . (($i - 1) * $per_page) . $append . '" title="' . $title .
521                                   '" class="pagination">' . $i . '</a>' . "\n";
522               }
523   
524               $pagination .= ' ';
525           }
526   
527           // Insert suffix ellipses if necessary
528           if ( $total_pages > 5 )
529           {
530               $pagination .= ( $end_count < $total_pages ) ? ' ... ' . "\n" : ' ';
531           }
532   
533           //--------------------------------
534           // Page n / Next Page links
535           //--------------------------------
536           if ( $on_page == $total_pages )
537           {
538               $pagination .= '<span style="font-weight: bold">' . $total_pages . '</span>' . "\n";
539           }
540           else
541           {
542               $title = sprintf($user->lang['PAGE_X'], $total_pages);
543               $title .= ' (' . ((($total_pages - 1) * $per_page) + 1) . ' - ' . ($num_items) . ')';
544   
545               $pagination .= '<a href="' . $base_url . '&amp;' . $start_variable . '=' . (($total_pages - 1) * $per_page) . $append .
546                               '" title="' . $title . '" class="pagination">' . $total_pages . '</a>' . "\n";
547   
548               $pagination .= ' <a href="' . $base_url . '&amp;' . $start_variable . '=' . ($on_page * $per_page) . $append .
549                               '" title="' . $user->lang['PAGE_NEXT'] . '" class="pagination">&gt;</a>' . "\n";
550           }
551   
552           return $pagination;
553       }

593       /**
594        * Check if a key has an entry in {@link $input}
595        *
596        * If multiple keys are being checked, method returns true if ANY ONE of
597        * them has input.
598        *
599        * @access public
600        * @param string|array Key(s) to check; Made into array if passed as string
601        * @return bool True if found, false if not
602        */
603       function has_input($keys)
604       {
605           if ( is_string($keys) )
606           {
607               $keys = array($keys);
608           }
609   
610           foreach ( $keys as $key )
611           {
612               if ( isset($this->input[$key]) )
613               {
614                   return true;
615               }
616           }
617   
618           return false;
619       }
