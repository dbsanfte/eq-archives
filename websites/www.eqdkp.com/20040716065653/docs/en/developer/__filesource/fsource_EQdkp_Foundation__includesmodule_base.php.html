41     * Base class for module frames.
42     *
43     * @abstract
44     * @author Rob Speicher <tsigo@eqdkp.com>

48    class Module_Base
49    {
50        /**
51         * Array of module data
52         * @access private
53         * @var array 
54         */
55        var $modules   = array();

58         * Array of templates to parse in Smarty
59         * @access private
60         * @var array 
61         */
62        var $templates = array();

71        /**
72         * Load a module group
73         *
74         * Populates the {@link $modules} array
75         *
76         * @access private
77         * @param string    Module group (ex: 'viewmember')
78         * @param bool      Call {@link init_group()} after loading?
79         * @return array {@link $modules}
80         */
81        function load_group($module_group, $init_group = true)
82        {
83            global $db
84    
85            $module_group = strtolower($module_group);
86    
87            $sql = 'SELECT m.module_code
88                    FROM ' . MODULES_TABLE . ' m, ' . MODULE_GROUPS_TABLE . " g
89                    WHERE (m.group_id = g.group_id)
90                    AND g.group_code = '" . $module_group . "'
91                    AND m.module_enabled = '1'
92                    ORDER BY m.module_order";
93            $result = $db->query($sql);
94            while ( $row = $db->fetch_record($result) )
95            {
96                $module_code  = strtolower($row['module_code']);
97    
98                // Put the name of each module class into an array
99                $this->modules[] = array(
100                   'GROUP'  => $module_group,
101                   'CODE'   => $module_code,
102                   'CLASS'  => ucfirst($module_group) . '_' . ucfirst($module_code),
103                   'OBJECT' => null,
104                   'START'  => array()
105               );
106           }
107           $db->free_result($result);
108   
109           if ( $init_group )
110           {
111               $this->init_group();
112           }
113   
114           return $this->modules;
115       }

117       /**
118        * Call {@link load_module()} and {@link init_module()} on a group of
119        * modules to save a developer time.
120        *
121        * Makes sure that the module isn't loaded twice.
122        *
123        * @access private
124        */
125       function init_group()
126       {
127           foreach ( $this->modules as $index => $module_data )
128           {
129               // Prevent double-loading
130               if ( $module_data['OBJECT'] == null )
131               {
132                   $this->load_module($index);
133                   $this->init_module($index);
134               }
135           }
136       }

138       /**
139        * Load a specific module
140        *
141        * Includes the module's class file but does not instantiate the class.
142        * Also adds the default template for the module to the {@link $templates}
143        * array.
144        *
145        * @access private
146        * @param int   Module's index in {@link $modules}
147        */
148       function load_module($index)
149       {
150           global $eqdkp
151   
152           $data = &$this->modules[$index];
153           $module_file = $eqdkp->root_path . 'modules/' . $data['GROUP'] . '/mod_' . $data['CODE'] . '.php';
154   
155           if ( file_exists($module_file) )
156           {
157               // Make the class available
158               include_once($module_file);
159           }
160           else
161           {
162               trigger_error('Failed to load ' . $data['GROUP'] . '/mod_' . $data['CODE'] . '.php', E_USER_ERROR);
163           }
164       }
165   
166       /**
167        * Initialize a specific module
168        *
169        * Makes sure that the module class is available and then instantiates it,
170        * storing the object in the {@link $modules} array.  Then it calls any
171        * initialization methods that the class has defined in its constructor
172        * using the {@link add_startmethod()} method.
173        *
174        * @access private
175        * @param int   Module's index in {@link $modules}
176        */
177       function init_module($index)
178       {
179           global $tpl $dbg $msg
180   
181           $data = &$this->modules[$index];
182   
183           if ( class_exists($data['CLASS']) )
184           {
185               // Instantiate the module, passing it a reference to this class and its child
186               $data['OBJECT'] = new $data['CLASS']($this, $index);
187   
188               // If we don't have any starting methods, null the object to save memory
189               if ( sizeof($data['START']) < 1 )
190               {
191                   $data['OBJECT'] = null;
192               }
193               else
194               {
195                   // This object is going to do something, so add a template for
196                   // it to use
197                   $this->add_template($index, 'mod_' . $data['CODE']);
198   
199                   // Call module's start method(s)
200                   $dbg->timer_start($data['CLASS']);
201                   foreach ( $data['START'] as $method )
202                   {
203                       $data['OBJECT']->$method();
204                   }
205                   $dbg->timer_stop($data['CLASS']);
206   
207                   // Assign module templates
208                   $tpl->assign('MODULES', $this->templates);
209   
210                   // Output any pending messages; if a module wants to stop output
211                   // it has to do it itself
212                   $msg->display(M_ERROR | M_MESSAGE | M_INSTRUCTION, false);
213               }
214           }
215           else
216           {
217               trigger_error('Undefined module class: ' . $data['CLASS'], E_USER_ERROR);
218           }
219       }
220   
221       /**
222        * Add method(s) to call when the class is instantiated to {@link $modules}
223        *
224        * @access private
225        * @param int             Module's index in {@link $modules}
226        * @param array|string   Name of method to add; if array, recurse
227        */
228       function add_startmethod($module_index, $method_name)
229       {
230           if ( is_array($method_name) )
231           {
232               foreach ( $method_name as $d_method_name )
233               {
234                   $this->add_startmethod($module_index, $d_method_name);
235               }
236           }
237           else
238           {
239               $this->modules[$module_index]['START'][] = $method_name;
240           }
241       }
242   
243       /**
244        * Add a template to parse
245        *
246        * @access private
247        * @param int       Module's index in {@link $modules}
248        * @param string    Template file to add
249        */
250       function add_template($module_index, $template)
251       {
252           $data = &$this->modules[$module_index];
253   
254           $template_file = 'modules/' . $data['GROUP'] . '/' . preg_replace('#(.+)\..+$#', '\1', $template);
255   
256           $this->templates[] = $template_file;
257       }

259       /**
260        * Dynamically load a module
261        *
262        * Load a specific module that may or may not be in a different group than
263        * the current group, without calling {@link load_group()}.  Necessary
264        * for loading of game modules, among other things.
265        *
266        * @access private
267        * @param string    Module group
268        * @param string    Module codename
269        */
270       function dyn_load($module_group, $module_code)
271       {
272           $module_group = strtolower($module_group);
273           $module_code  = strtolower($module_code);
274   
275           $this->modules[] = array(
276               'GROUP'  => $module_group,
277               'CODE'   => $module_code,
278               'CLASS'  => ucfirst($module_group) . '_' . ucfirst($module_code),
279               'OBJECT' => null,
280               'START'  => array()
281           );
282   
283           $index = sizeof($this->modules) - 1;
284           $this->load_module($index);
285       }

287       /**
288        * Load the current game module for a module to use
289        *
290        * @access public
291        * @return object|null Game Module's object
292        */
293       function load_gamemod()
294       {
295           global $eqdkp
296   
297           if ( !class_exists('Game_Module') )
298           {
299               $this->dyn_load('games', $eqdkp->config['gamemod']);
300               $game = new Game_Module();
301   
302               return $game;
303           }
304   
305           return null;
306       }
