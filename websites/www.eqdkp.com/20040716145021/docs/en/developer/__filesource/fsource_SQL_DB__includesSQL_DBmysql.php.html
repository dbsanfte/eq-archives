145       /**
146        * Connect to a database.
147        *
148        * If a link already exists with the same {@link $dbhost},
149        * {@link $dbuser} and {@link $dbpass} info, then {@link select_db()} is
150        * called instead of creating a new connection.
151        *
152        * @access public
153        * @param string    Database host
154        * @param string    Database name
155        * @param string    Database username
156        * @param string    Database password
157        * @param bool      Persistant connection?
158        * @return bool|resourceFalse if it couldn't connect, link_id if it did
159        */
160       function connect($dbhost, $dbname, $dbuser, $dbpass='', $pconnect = false)
161       {
162           // If we're connecting with the same host/user/pass as the one we already have,
163           // simply select the database rather than reconnecting to it
164           if ( ($this->dbhost == $dbhost) && ($this->dbuser == $dbuser) && ($this->dbpass == $dbpass) )
165           {
166               $previous_db = $this->dbname;
167               $this->select_db($dbname);
168   
169               return $previous_db;
170           }
171           // Otherwise, make a new connection
172           else
173           {
174               $this->pconnect = $pconnect;
175               $this->dbhost   = $dbhost;
176               $this->dbname   = $dbname;
177               $this->dbuser   = $dbuser;
178               $this->dbpass   = $dbpass;
179   
180               //--------------------------------
181               // Persistent connection
182               //--------------------------------
183   
184               if ( $this->pconnect )
185               {
186                   if ( empty($this->dbpass) )
187                   {
188                       $this->link_id = mysql_pconnect($this->dbhost, $this->dbuser);
189                   }
190                   else
191                   {
192                       $this->link_id = mysql_pconnect($this->dbhost, $this->dbuser, $this->dbpass);
193                   }
194               }
195   
196               //--------------------------------
197               // Regular connection
198               //--------------------------------
199   
200               else
201               {
202                   if ( empty($this->dbpass) )
203                   {
204                       $this->link_id = mysql_connect($this->dbhost, $this->dbuser);
205                   }
206                   else
207                   {
208                       $this->link_id = mysql_connect($this->dbhost, $this->dbuser, $this->dbpass);
209                   }
210               }
211   
212               if ( (is_resource($this->link_id)) && (!is_null($this->link_id)) && ($this->dbname != '') )
213               {
214                   return $this->select_db($this->dbname);
215               }
216               else
217               {
218                   return false;
219               }
220           }
221       }

255       /**
256        * Perform a database query
257        *
258        * @access public
259        * @param string    Query to run
260        * @return string|resourceReturn error message or resource of query result
261        */
262       function query($query)
263       {
264           // Remove pre-existing query resources
265           unset($this->query_id);
266   
267           if ( $query != '' )
268           {
269               $this->query_count++;
270               $this->query_id = mysql_query($query, $this->link_id);
271           }
272   
273           if ( !empty($this->query_id) )
274           {
275               $this->queries[$this->query_count] = $query;
276   
277               unset($this->record[$this->query_id]);
278               unset($this->record_set[$this->query_id]);
279               return $this->query_id;
280           }
281           else
282           {
283               $error = $this->error();
284               $message  = 'SQL query error<br /><br />';
285               $message .= 'Query: '.$query.'<br />';
286               $message .= 'Message: '.$error['message'].'<br />';
287               $message .= 'Code: '.$error['code'];
288   
289               if ( $this->error_die )
290               {
291                   die($message);
292               }
293   
294               return $message;
295           }
296       }

298       /**
299        * Return one row from a query
300        *
301        * @access public
302        * @param string    Query to run
303        * @return bool|mixedFalse if the query returned nothing, otherwise return the first record
304        */
305       function query_first($query)
306       {
307           $this->query($query);
308           $record = $this->fetch_record($this->query_id, MYSQL_BOTH);
309           $this->free_result($this->query_id);
310   
311           return ( isset($record[0]) ) ? $record[0] : false;
312       }

314       /**
315        * Build a query to use in an INSERT or UPDATE query
316        *
317        * @access public
318        * @param string    Query type ('INSERT'|'UPDATE')
319        * @param array     field => value pairs to use in query
320        * @return bool|stringFalse if couldn't create the query, otherwise returns the query
321        */
322       function build_query($query, $array = false)
323       {
324           if ( !is_array($array) )
325           {
326               return false;
327           }
328   
329           $fields = array();
330           $values = array();
331   
332           if ( $query == 'INSERT' )
333           {
334               foreach ( $array as $field => $value )
335               {
336                   $fields[] = $field;
337   
338                   if ( is_null($value) )
339                   {
340                       $values[] = 'NULL';
341                   }
342                   elseif ( is_string($value) )
343                   {
344                       $values[] = "'" . $this->escape($value) . "'";
345                   }
346                   else
347                   {
348                       $values[] = ( is_bool($value) ) ? intval($value) : $value;
349                   }
350               }
351   
352               $query = ' (' . implode(', ', $fields) . ') VALUES (' . implode(', ', $values) . ')';
353           }
354           elseif ( $query == 'UPDATE' )
355           {
356               foreach ( $array as $field => $value )
357               {
358                   if ( is_null($value) )
359                   {
360                       $values[] = "$field = NULL";
361                   }
362                   elseif ( is_string($value) )
363                   {
364                       $values[] = "$field = '" . $this->escape($value) . "'";
365                   }
366                   else
367                   {
368                       $values[] = ( is_bool($value) ) ? "$field = " . intval($value) : "$field = $value";
369                   }
370               }
371   
372               $query = implode(', ', $values);
373           }
374   
375           return $query;
376       }

378       /**
379        * Fetch a specific record
380        *
381        * @access public
382        * @param resource  Query ID resource
383        * @param int       MYSQL_ASSOC, MYSQL_BOTH or MYSQL_NUM
384        * @return bool|arrayFalse or record array
385        */
386       function fetch_record($query_id = 0, $type = MYSQL_ASSOC)
387       {
388           if ( !$query_id )
389           {
390               $query_id = $this->query_id;
391           }
392   
393           if ( $query_id )
394           {
395               $this->record[$query_id] = @mysql_fetch_array($query_id, $type);
396               return $this->record[$query_id];
397           }
398           else
399           {
400               return false;
401           }
402       }

404       /**
405        * Fetch a group of records
406        *
407        * @access public
408        * @param resource  Query ID resource
409        * @return bool|resourceFalse or record array
410        */
411       function fetch_record_set($query_id = 0)
412       {
413           if ( !$query_id )
414           {
415               $query_id = $this->query_id;
416           }
417           if ( $query_id )
418           {
419               unset($this->record_set[$query_id]);
420               unset($this->record[$query_id]);
421               while ( $this->record_set[$query_id] = @mysql_fetch_array($query_id) )
422               {
423                   $result[] = $this->record_set[$query_id];
424               }
425               return $result;
426           }
427           else
428           {
429               return false;
430           }
431       }

488       /**
489        * Free a MySQL result resource
490        *
491        * @access public
492        * @param resource  Query ID resource
493        * @return bool 
494        */
495       function free_result($query_id = 0)
496       {
497           if ( $query_id == 0 )
498           {
499               $query_id = $this->query_id;
500           }
501   
502           if ( is_resource($query_id) )
503           {
504               mysql_free_result($query_id);
505   
506               unset($this->record[$query_id]);
507               unset($this->record_set[$query_id]);
508   
509               return true;
510           }
511           else
512           {
513               return false;
514           }

522        * @return string Escaped string

524       function escape($string)
525       {
526           $string = str_replace("'", "''",    $string);

528   
529           return $string;
