60        /**
61         * Get a file list
62         *
63         * @access public
64         * @param string    Directory to get file list from
65         * @param bool      Recurse through sub-directories?
66         * @return array    Array of file names
67         */
68        function file_list($directory, $recurse = false)
69        {
70            if ( $dir = @opendir($directory) )
71            {
72                while ( $file = @readdir($dir) )
73                {
74                    if ( (is_file($directory . '/' . $file)) && ($file != '.') && ($file != '..') && (strtoupper($file) != 'CVS') )
75                    {
76                        $this->files[$directory][] = $file;
77                    }
78                    elseif ( (is_dir($directory . '/' . $file)) && ($file != '.') && ($file != '..') && (strtoupper($file) != 'CVS') && ($recurse) )
79                    {
80                        $this->file_list($directory . $file, $recurse);
81                    }
82                }
83            }
84    
85            return $this->files;
86        }
