50    /**
51     * Raid History module for ViewMember
52     *
53     * @author Rob Speicher <tsigo@eqdkp.com>
54     * @package EQdkp_User
55     * @subpackage ViewMember
56     * @version $Id: mod_raid_history.php,v 1.3 2003/12/23 22:42:20 tsigo Exp $
57     */
58    class Viewmember_Raid_history
59    {
60        /**
61         * Reference to base class object that loaded this module
62         * @access private
63         * @var object ref 
64         */
65        var $loader = null;
66    
67        /**
68         * Location of this module in the Module_Base::modules array
69         * @access private
70         * @var int 
71         */
72        var $index  = -1;
73    
74        /**
75         * Total number of raids
76         * @access private
77         * @var int 
78         */
79        var $num_raids = 0;
80    
81        /**
82         * Pagination start variable
83         * @access private
84         * @var string 
85         */
86        var $start_var = 'rstart';
87    
88        /**
89         * Pagination start number
90         * @access private
91         * @var int 
92         */
93        var $start     = 0;
94    
95        /**
96         * Set the values of {@link $loader} and {@link $index},
97         * add the module's language pack and tell Module_Base this module's
98         * initialization methods.
99         *
100        * @param object ref Setting for {@link $loader}
101        * @param int        Setting for {@link $index}
102        */
103       function viewmember_raid_history(&$module_loader, $module_index)
104       {
105           global $user $msg
106   
107           $this->loader = $module_loader;
108           $this->index  = $module_index;
109   
110           $user->add_lang('modules/viewmember/', 'raid_history');
111   
112           //--------------------------------
113           // Check permissions
114           //--------------------------------
115   
116           if ( !$user->auth(AG_RAID, AO_LIST) )
117           {
118               $msg->add(M_ERROR, $user->lang['E_LISTRAIDS_DENIED']);
119           }
120           else
121           {
122               $this->loader->add_startmethod($this->index, array(
123                   'raid_rows',
124                   'footer')
125               );
126           }
127       }
128   
129       /**
130        * Output raid rows
131        *
132        * @access protected
133        */
134       function raid_rows()
135       {
136           global $eqdkp $user $db $tpl
137           global $SID
138   
139           $this->start = ( isset($eqdkp->input[$this->start_var]) ) ? intval($eqdkp->input[$this->start_var]) : 0;
140   
141           //--------------------------------
142           // Calculate the 'Current' earned column based on the paging start
143           //--------------------------------
144   
145           $current_earned = $this->loader->member['earned'];
146           if ( $this->start > 0 )
147           {
148               $sql = 'SELECT raid_value
149                       FROM ' . RAIDS_TABLE . ' r, ' . RAID_ATTENDEES_TABLE . " ra
150                       WHERE (r.raid_id = ra.raid_id)
151                       AND ra.member_name = '" . $this->loader->member['name'] . "'
152                       ORDER BY r.raid_date DESC
153                       LIMIT " . $this->start;
154               $result = $db->query($sql);
155               while ( $row = $db->fetch_record($result) )
156               {
157                   $current_earned -= $row['raid_value'];
158               }
159               $db->free_result($result);
160           }
161   
162           //--------------------------------
163           // Output rows
164           //--------------------------------
165   
166           $sql = 'SELECT r.raid_id, r.raid_name, r.raid_date, r.raid_note, r.raid_value
167                   FROM ' . RAIDS_TABLE . ' r, ' . RAID_ATTENDEES_TABLE . " ra
168                   WHERE (r.raid_id = ra.raid_id)
169                   AND ra.member_name = '" . $this->loader->member['name'] . "'
170                   ORDER BY r.raid_date DESC
171                   LIMIT " . $this->start . ',' . $user->data['user_rlimit'];
172           $result = $db->query($sql);
173           while ( $row = $db->fetch_record($result) )
174           {
175               $tpl->append_block('RAIDS_ROW', array(
176                   'U_VIEWRAID' => 'viewraid.php' . $SID . '&amp;' . URI_RAID . '=' . $row['raid_id'],
177                   'U_EDITRAID' => 'admin/addraid.php' . $SID . '&amp;' . URI_RAID . '=' . $row['raid_id'],
178                   'ID'         => $row['raid_id'],
179                   'DATE'       => $eqdkp->date($user->style['date_notime_short'], $row['raid_date']),
180                   'NAME'       => ( $row['raid_name'] != '' ) ? stripslashes($row['raid_name']) : '&nbsp;',
181                   'NOTE'       => ( $row['raid_note'] != '' ) ? stripslashes($row['raid_note']) : '&nbsp;',
182                   'EARNED'     => $row['raid_value'],
183                   'CURRENT'    => $current_earned)
184               );
185   
186               $current_earned -= $row['raid_value'];
187           }
188           $db->free_result($result);
189   
190           $tpl->assign_block('RAIDS_ROW');
191       }
192   
193       /**
194        * Output raid pagination
195        *
196        * @access protected
197        */
198       function footer()
199       {
200           global $eqdkp $user $db $tpl
201   
202           $sql = 'SELECT count(r.raid_id)
203                   FROM ' . RAIDS_TABLE . ' r, ' . RAID_ATTENDEES_TABLE . " ra
204                   WHERE (r.raid_id = ra.raid_id)
205                   AND ra.member_name = '" . $this->loader->member['name'] . "'";
206           $this->num_raids = $db->query_first($sql);
207   
208           $tpl->assign(array(
209               'NUM_RAIDS'       => $this->num_raids,
210               'RAID_PAGINATION' => $eqdkp->pagination($this->num_raids, $user->data['user_rlimit'], $this->start, $this->start_var))
211           );
212       }
