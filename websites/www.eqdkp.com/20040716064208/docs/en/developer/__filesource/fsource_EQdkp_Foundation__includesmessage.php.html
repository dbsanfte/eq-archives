40    /** @ignore */
41    define('M_ERROR',       1);
42    define('M_MESSAGE',     2);
43    define('M_INSTRUCTION', 4);
44    
45    /**
46     * Messaging system
47     *
48     * Handles printing out error messages, general messages, and user instructions.
49     *
50     * @author Rob Speicher <tsigo@eqdkp.com>
51     * @package EQdkp_Foundation
52     * @version $Id: message.php,v 1.2 2003/12/22 06:18:56 tsigo Exp $
53     */
54    class Message
55    {
56        /**#@+
57         * @access private
58         * @var array 
59         */
60        var $error       = array();
61        var $message     = array();
62        var $instruction = array();
63        /**#@-*/
64    
65        /**
66         * @access private
67         * @var bool 
68         */
69        var $clear_output = null;
70    
71        /**
72         * @ignore
73         */
74        function message()
75        {
76        }
77    
78        /**
79         * Set the value of {@link $clear_output}
80         *
81         * If set (true or false, not null), it will overwrite the second parameter
82         * of {@link display()}.
83         *
84         * @access public
85         * @param bool 
86         */
87        function clear_output($setting = null)
88        {
89            $this->clear_output = $setting;
90        }
91    
92        /**
93         * Add a message
94         *
95         * @access public
96         * @param int       Type of message to add (M_ERROR, M_MESSAGE, M_INSTRUCTION)
97         * @param string    Message to add
98         */
99        function add($type, $text)
100       {
101           switch ( $type )
102           {
103               case M_ERROR:
104                   $this->add_error($text);
105                   break;
106               case M_MESSAGE:
107                   $this->add_message($text);
108                   break;
109               case M_INSTRUCTION:
110                   $this->add_instruction($text);
111                   break;
112           }
113       }
114   
115       /**
116        * Add an error message
117        *
118        * @access private
119        * @param string    Message to add
120        */
121       function add_error($text)
122       {
123           $this->error[] = $text;
124       }
125   
126       /**
127        * Add a general message
128        *
129        * @access private
130        * @param string    Message to add
131        */
132       function add_message($text)
133       {
134           $this->message[] = $text;
135       }
136   
137       /**
138        * Add an instructional message
139        *
140        * @access private
141        * @param string    Message to add
142        */
143       function add_instruction($text)
144       {
145           $this->instruction[] = $text;
146       }
147   
148       /**
149        * Display requested type(s) of messages
150        *
151        * $type can be |'d together to output multiple types
152        *
153        * @access public
154        * @param int       Type of message(s) to output
155        * @param bool      Hide any other output?
156        */
157       function display($type, $clear_output = false)
158       {
159           global $eqdkp $user $tpl
160   
161           $clear_output = ( isset($this->clear_output) ) ? $this->clear_output : $clear_output;
162   
163           if ( ($type & M_ERROR) && (sizeof($this->error) > 0) )
164           {
165               $tpl->append(array(
166                   'MESSAGES' => true,
167                   'M_ERROR'  => $this->error), null, true
168               );
169   
170               $this->error = array();
171           }
172   
173           if ( ($type & M_MESSAGE) && (sizeof($this->message) > 0) )
174           {
175               $tpl->append(array(
176                   'MESSAGES'  => true,
177                   'M_MESSAGE' => $this->message), null, true
178               );
179   
180               $this->message = array();
181           }
182   
183           if ( ($type & M_INSTRUCTION) && (sizeof($this->instruction) > 0) )
184           {
185               $tpl->append(array(
186                   'MESSAGES'       => true,
187                   'M_INSTRUCTION'  => $this->instruction), null, true
188               );
189   
190               $this->instruction = array();
191           }
192   
193           // Prevent other output from being shown, if requested
194           if ( $clear_output )
195           {
196               $tpl->set_vars(array(
197                   'page_title'    => $eqdkp->title($user->lang['MESSAGE']),
198                   'out_types'     => array('html'),
199                   'template_file' => 'message.html',
200                   'display'       => true)
201               );
202           }
203       }
204   }
