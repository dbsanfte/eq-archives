40    /**
41     * Debug class
42     *
43     * Keeps track of an infinite number of timers
44     *
45     * @author Rob Speicher <tsigo@eqdkp.com>
46     * @package EQdkp_Foundation
47     * @version $Id: debug.php,v 1.1.1.1 2003/12/21 22:26:32 tsigo Exp $
48     */
49    class Debug
50    {
51        /**#@+
52         * @access private
53         * @var array 
54         */
55        var $timer_starts = array();
56        var $timer_stops  = array();
57        /**#@-*/
58    
59        /**
60         * @ignore
61         */
62        function debug()
63        {
64        }
65    
66        /**
67         * Start a new timer
68         *
69         * @access public
70         * @param string    Index of the timer; Defaults to root (index 0) or array size + 1
71         * @return string Index of the timer, in case none was provided
72         */
73        function timer_start($timer_index = '')
74        {
75            // Timer index becomes 'root' if it's our first timer, otherwise it's the size of the array + 1
76            if ( $timer_index == '' )
77            {
78                $timer_index = ( isset($this->timer_starts['root']) ) ? sizeof($this->timer_starts) + 1 : 'root';
79            }
80    
81            // Tickety-tock-tickety...
82            $mc_split = split(' ', microtime());
83            $this->timer_starts[$timer_index] = $mc_split[0] + $mc_split[1];
84            unset($mc_split);
85    
86            return $timer_index;
87        }
88    
89        /**
90         * Stop a timer
91         *
92         * Returns nothing, but inserts a value in the {@link $timer_stops} array.
93         *
94         * @access private
95         * @param string    Index of the timer
96         */
97        function timer_stop($timer_index)
98        {
99            $mc_split = split(' ', microtime());
100           $this->timer_stops[$timer_index] = $mc_split[0] + $mc_split[1];
101           unset($mc_split);
102       }
103   
104       /**
105        * Find out the time elapsed between start and stop.
106        *
107        * @access public
108        * @param string    Index of the timer
109        * @param int       How many digits to return from the elapsed time
110        * @return string   Elapsed time
111        */
112       function timer_elapsed($timer_index, $precision = 5)
113       {
114           if ( !isset($this->timer_stops[$timer_index]) )
115           {
116               $this->timer_stop($timer_index);
117           }
118   
119           $timer_elapsed = $this->timer_stops[$timer_index] - $this->timer_starts[$timer_index];
120           $timer_elapsed = substr($timer_elapsed, 0, $precision);
121   
122           return $timer_elapsed;
123       }
124   
125       /**
126        * Find out the time elapsed between start and stop for all timers
127        *
128        * @access public
129        * @param int   How many digits to return from the elapsed time
130        * @return array Array of arrays containing INDEX and ELAPSED data
131        */
132       function timers_elapsed($precision = 5)
133       {
134           $elapsed = array();
135           foreach ( $this->timer_starts as $timer_index => $timer_start )
136           {
137               $elapsed[$timer_index] = array(
138                   'INDEX'   => $timer_index,
139                   'ELAPSED' => $this->timer_elapsed($timer_index, $precision)
140               );
141           }
142   
143           return $elapsed;
144       }
145   }
