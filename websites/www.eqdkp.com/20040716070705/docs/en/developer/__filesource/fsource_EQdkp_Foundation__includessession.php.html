75        /**
76         * Initializes {@link $address} and {@link $location} values
77         */
78        function session()
79        {
80            global $eqdkp
81    
82            //--------------------------------
83            // Set up our vars
84            //--------------------------------
85    
86            $this->address  = ( !empty($_SERVER['REMOTE_ADDR']) ) ? $_SERVER['REMOTE_ADDR'] : $REMOTE_ADDR;
87            $this->location = $eqdkp->uri_self(false, URI_SESSION);
88            $this->location = preg_replace('#^.*\/(.*)$#', '\1', $this->location);
89        }
90    
91        /**
92         * Call {@link cleanup()}, check for banned IPs, and call
93         * {@link create_guest()}, {@link update_guest()}, {@link create_user()},
94         * or {@link update_user()} to create or update the session.
95         *
96         * @access public
97         * @return string Session ID URL string
98         */
99        function start()
100       {
101           global $db
102   
103           //--------------------------------
104           // Remove dead sessions before we get started
105           //--------------------------------
106   
107           $this->cleanup();
108   
109           //--------------------------------
110           // Check for banned IPs
111           //--------------------------------
112   
113           //--------------------------------
114           // Check for an existing session from this IP
115           //--------------------------------
116   
117           $sql = 'SELECT session_id
118                   FROM ' . SESSIONS_TABLE . "
119                   WHERE session_address = '" . $this->address . "'";
120           $result = $db->query($sql);
121           if ( $db->num_rows($result) > 0 )
122           {
123               // At least one session from this IP exists
124               $row = $db->fetch_record($result);
125               $this->update_guest($row['session_id']);
126           }
127           else
128           {
129               // No previous session from this IP, create a new one
130               $this->create_guest();
131           }
132           $db->free_result($result);
133   
134           //--------------------------------
135           // Session has been registered and we should have a session id
136           //--------------------------------
137   
138           $SID = '?' . URI_SESSION . '=' . $this->session_id;
139   
140           return $SID;
141       }

143       /**
144        * Remove dead sessions from session table
145        *
146        * Called by {@link start()} before session is created or updated
147        *
148        * @access private
149        */
150       function cleanup()
151       {
152           global $eqdkp $db
153   
154           //--------------------------------
155           // Remove dead sessions, as deemed by session_length
156           //--------------------------------
157   
158           $session_cutoff = $eqdkp->config['session_last_cleanup'] - $eqdkp->config['session_length'];
159           $sql = 'DELETE FROM ' . SESSIONS_TABLE . '
160                   WHERE session_updated < ' . $session_cutoff;
161           $result = $db->query($sql);
162           $db->free_result($result);
163   
164           //--------------------------------
165           // Update the session_last_cleanup setting
166           //--------------------------------
167   
168           $eqdkp->config_set('session_last_cleanup', time());
169       }

171       /**
172        * Create a new guest session
173        *
174        * @access private
175        */
176       function create_guest()
177       {
178           global $db
179   
180           $this->session_id = md5(uniqid($this->address, true));
181   
182           //--------------------------------
183           // Insert an anonymous user session
184           //--------------------------------
185   
186           $query = $db->build_query('INSERT', array(
187               'session_id'       => $this->session_id,
188               'user_id'          => ANONYMOUS,
189               'session_created'  => time(),
190               'session_updated'  => time(),
191               'session_location' => $this->location,
192               'session_address'  => $this->address)
193           );
194           $db->query('INSERT INTO ' . SESSIONS_TABLE . $query);
195   
196           //--------------------------------
197           // Provide guest data values not provided by User class
198           //--------------------------------
199   
200           $this->data = array(
201               'user_id'  => ANONYMOUS,
202               'username' => '');
203       }
204   
205       /**
206        * Update an existing guest session
207        *
208        * @access private
209        * @param string    Session ID to update
210        */
211       function update_guest($session_id)
212       {
213           global $db
214   
215           $this->session_id = $session_id;
216   
217           //--------------------------------
218           // Update an anonymous session
219           //--------------------------------
220   
221           $query = $db->build_query('UPDATE', array(
222               'user_id'          => ANONYMOUS,
223               'session_updated'  => time(),
224               'session_location' => $this->location)
225           );
226           $db->query('UPDATE ' . SESSIONS_TABLE . ' SET ' . $query . " WHERE session_id = '" . $session_id . "'");
227   
228           //--------------------------------
229           // Provide guest data values not provided by User class
230           //--------------------------------
231   
232           $this->data = array(
233               'user_id'  => ANONYMOUS,
234               'username' => '');
235       }

293       /**
294        * Set up the user's {@link $lang} array, set the user's locale,
295        * get the user's {@link $style} data, and get the user's {@link $auth}
296        * data.
297        *
298        * @access public
299        * @param bool|int Style ID to use instead of user's default
300        */
301       function setup($style_id = false)
302       {
303           global $eqdkp $db $tpl
304   
305           //--------------------------------
306           // Set up language paths
307           //--------------------------------
308   
309           if ( (isset($this->data['user_id'])) && ($this->data['user_id'] != ANONYMOUS) && (!empty($this->data['user_locale'])) )
310           {
311               setlocale(LC_ALL, $this->data['user_locale']);
312   
313               $this->lang_name = ( file_exists($eqdkp->root_path . 'language/' . strtolower($this->data['user_locale'])) ) ? strtolower($this->data['user_locale'])
314                                                                                                                            : strtolower($eqdkp->config['default_locale']);
315               $this->lang_path = $eqdkp->root_path . 'language/' . $this->lang_name . '/';
316           }
317           else
318           {
319               setlocale(LC_ALL, $eqdkp->config['default_locale']);
320   
321               $this->lang_name = strtolower($eqdkp->config['default_locale']);
322               $this->lang_path = $eqdkp->root_path . 'language/' . $this->lang_name . '/';
323           }
324   
325           //--------------------------------
326           // Set locale info
327           //--------------------------------
328   
329           $this->locale = localeconv();
330   
331           //--------------------------------
332           // Get main language pack
333           //--------------------------------
334   
335           include_once($this->lang_path . 'lang_main.php');
336           $this->lang = ( isset($lang) ) ? $lang : array();
337   
338           //--------------------------------
339           // Set up style
340           //--------------------------------
341   
342           $style_id = ( $style_id ) ? $style_id : (( $this->data['user_id'] != ANONYMOUS) ? $this->data['user_style'] : $eqdkp->config['default_style']);
343   
344           $sql = 'SELECT s.*, c.*
345                   FROM ' . STYLES_TABLE . ' s, ' . STYLE_CONFIG_TABLE . ' c
346                   WHERE s.style_id = c.style_id
347                   AND s.style_id = ' . $style_id;
348           $result = $db->query($sql);
349           $this->style = $db->fetch_record($result);
350           $db->free_result($result);
351   
352           $tpl->set_template($this->style['template_path']);
353   
354           //--------------------------------
355           // Default limits if user is anonymous
356           //--------------------------------
357   
358           if ( $this->data['user_id'] == ANONYMOUS )
359           {
360               $this->data['user_locale'] = $eqdkp->config['default_locale'];
361               $this->data['user_style']  = $eqdkp->config['default_style'];
362               $this->data['user_alimit'] = $eqdkp->config['default_alimit'];
363               $this->data['user_elimit'] = $eqdkp->config['default_elimit'];
364               $this->data['user_ilimit'] = $eqdkp->config['default_ilimit'];
365               $this->data['user_nlimit'] = $eqdkp->config['default_nlimit'];
366               $this->data['user_rlimit'] = $eqdkp->config['default_rlimit'];
367           }
368   
369           //--------------------------------
370           // Auth settings
371           //--------------------------------
372   
373           /** TEMP **/
374           $this->auth = array(
375               AG_ADJUSTMENT => (AO_LIST | AO_VIEW | AO_ADD | AO_UPDATE | AO_DELETE | AO_MANAGE),
376               AG_EVENT      => (AO_LIST | AO_VIEW | AO_ADD | AO_UPDATE | AO_DELETE | AO_MANAGE),
377               AG_ITEM       => (AO_LIST | AO_VIEW | AO_ADD | AO_UPDATE | AO_DELETE | AO_MANAGE),
378               AG_MEMBER     => (AO_LIST | AO_VIEW | AO_ADD | AO_UPDATE | AO_DELETE | AO_MANAGE),
379               AG_NEWS       => (AO_LIST | AO_VIEW | AO_ADD | AO_UPDATE | AO_DELETE | AO_MANAGE),
380               AG_RAID       => (AO_LIST | AO_VIEW | AO_ADD | AO_UPDATE | AO_DELETE | AO_MANAGE),
381               AG_STYLE      => (AO_LIST | AO_VIEW | AO_ADD | AO_UPDATE | AO_DELETE | AO_MANAGE)
382           );
383       }
384   
385       /**
386        * Add a specific language pack to the {@link $lang} array
387        *
388        * @access public
389        * @param string    Path to the pack, relative to language/locale/
390        * @param string    Specific pack to load ('lang_$pack.php')
391        * @return bool     Whether or not pack was added
392        */
393       function add_lang($path, $pack)
394       {
395           if ( file_exists($this->lang_path . $path . 'lang_' . $pack . '.php') )
396           {
397               include($this->lang_path . $path . 'lang_' . $pack . '.php');
398   
399               if ( isset($lang) )
400               {
401                   $this->lang = array_merge($this->lang, $lang);
402                   unset($lang);
403   
404                   return true;
405               }
406           }
407   
408           return false;
409       }

411       /**
412        * Check if a user has permission to do something
413        *
414        * @access public
415        * @param int|string   Permission group to check in
416        * @param int|string   Permission option(s) to check
417        * @param int           ID of user to check, if not the current
418        * @return bool         User does or does not have permission
419        */
420       function auth($auth_group, $auth_option = AO_ANY, $user_id = 0)
421       {
422           // First make sure we're using the actual constant value and not
423           // the string name of that constant
424           if ( is_string($auth_group) )
425           {
426               $auth_group = constant($auth_group);
427           }
428           if ( is_string($auth_option) )
429           {
430               $auth_option = constant($auth_option);
431           }
432   
433           // Are we looking for any permissions from this group?
434           if ( $auth_option == AO_ANY )
435           {
436               if ( $this->auth[$auth_group] > AO_ANY )
437               {
438                   return true;
439               }
440               else
441               {
442                   return false;
443               }
444           }
445           // No, we're looking for a specific permission
446           else
447           {
448               if ( $this->auth[$auth_group] & $auth_option )
449               {
450                   return true;
451               }
452               else
453               {
454                   return false;
455               }
456           }
457       }
