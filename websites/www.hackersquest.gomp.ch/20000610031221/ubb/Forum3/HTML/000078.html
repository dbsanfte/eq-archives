code:

<HTML>
<TITLE>EQ Experience Calculator</TITLE>
<HEAD>
<SCRIPT LANGUAGE="JavaScript">
var raceName = null;
var raceMult = 0;
var className = null;
var classMult = 0;
function doTable(){
        var cr = raceName + " " + className;
        var base = raceMult * classMult ;
        document.open();
        document.write("<html><head><title>XP vs Level for "+cr+" (base = "+base+")</title></head>");
        document.write("<body><center><h3>"+cr+" ("+base+")</h3>");
        document.write("<table border width=95%><tr>");
        for (var l=1;l<6;l++) {
                document.write("<td align=right>Level</td><td width=20 align=right>XP needed</td>");
        }
        document.write("</tr>");
        for (var x=1;x<11;x++) {
           document.write("<tr>");
           for (var l=x; l<51; l += 10) {
                var b=Math.floor(((l>30) ? 1+Math.floor((l-30+4)/5)*0.1 : 1)*base*Math.pow(l-1,3));
                b += ""; // make it a string...
                var bs = ""
                while (b.length>3) {
                        u = b.length;
                        bs = "," + b.slice(u-3,u) + bs;
                        b = b.slice(0,u-3);
                }
                bs = b + bs ;
                document.write("<td align=right>"+l+"</td><td align=right>"+bs+"</td>");
           }
           document.write("</tr>");               
        }
        document.write("</tr></table>");
        document.write("<a href="+parent.location+">another</a><p>");
        document.write("</center></body></html>");
        document.close();
}
function changedRace(what) {
    if (what.selectedIndex>0) {
                raceMult = what.options[what.selectedIndex].value;
                raceName = what.options[what.selectedIndex].text;
                if (classMult > 0) {
                        doTable();
                }
    }
}
function changedClass(what) {
        if (what.selectedIndex>0) {
                classMult = what.options[what.selectedIndex].value;
                className = what.options[what.selectedIndex].text;
                if (raceMult > 0) {
                        doTable();
                }
        }
}
</SCRIPT>
</HEAD>
