Depending on the window you use, your item naming conventions may be different. Also I posted using a custom gauge, so your gauge background, gaugefill and gaugelinefill will be different.


<Gauge item = "GW_Player_ExpGauge">
<ScreenID>ExpGauge</ScreenID>
<!--<Font>3</Font>-->
<RelativePosition>true</RelativePosition>
<Location>
<X>7</X>
<Y>93</Y>
</Location>
<Size>
<CX>101</CX>
<CY>12</CY>
</Size>
<GaugeOffsetY>0</GaugeOffsetY>
<Style_VScroll>false</Style_VScroll>
<Style_HScroll>false</Style_HScroll>
<Style_Transparent>false</Style_Transparent>
<!--<TooltipReference/>-->
<FillTint>
<R>220</R>
<G>150</G>
<B>0</B>
</FillTint>
<LinesFillTint>
<R>40</R>
<G>40</G>
<B>255</B>
</LinesFillTint>
<DrawLinesFill>true</DrawLinesFill>
<EQType>4</EQType>
<GaugeDrawTemplate>
<Background>A_BigGaugeBackground</Background>
<Fill>A_BigGaugeFill</Fill>
<LinesFill>A_BigGaugeLinesFill</LinesFill>
</GaugeDrawTemplate>
</Gauge>
