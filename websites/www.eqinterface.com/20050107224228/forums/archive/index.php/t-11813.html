<Gauge item = "Target_EmptyCasting_Gauge">
<ScreenID>CastGauge</ScreenID>
<!--<Font>3</Font> -->
<RelativePosition>true</RelativePosition>
<Location>
<X>130</X>
<Y>39</Y>
</Location>
<Size>
<CX>116</CX>
<CY>8</CY>
</Size>
<GaugeOffsetY>0</GaugeOffsetY>
<GaugeOffsetX>0</GaugeOffsetX>
<Style_VScroll>false</Style_VScroll>
<Style_HScroll>false</Style_HScroll>
<Style_Transparent>false</Style_Transparent>
<!--<TooltipReference/>-->
<GaugeDrawTemplate>
<Background>A_GaugeBackground</Background>
<EndCapLeft>A_GaugeEndCapLeft</EndCapLeft>
<EndCapRight>A_GaugeEndCapRight</EndCapRight>
</GaugeDrawTemplate>
</Gauge>

<Gauge item = "Target_Casting_Gauge">
<ScreenID>CastGauge</ScreenID>
<!--<Font>3</Font> -->
<RelativePosition>true</RelativePosition>
<Location>
<X>130</X>
<Y>39</Y>
</Location>
<Size>
<CX>116</CX>
<CY>8</CY>
</Size>
<GaugeOffsetY>0</GaugeOffsetY>
<GaugeOffsetX>0</GaugeOffsetX>
<Style_VScroll>false</Style_VScroll>
<Style_HScroll>false</Style_HScroll>
<Style_Transparent>false</Style_Transparent>
<!--<TooltipReference/>-->
<FillTint>
<R>240</R>
<G>0</G>
<B>240</B>
</FillTint>
<LinesFillTint>
<R>0</R>
<G>0</G>
<B>0</B>
</LinesFillTint>
<DrawLinesFill>false</DrawLinesFill>
<EQType>7</EQType>
<GaugeDrawTemplate>
<Background>A_GaugeBackground</Background>
<Fill>A_GaugeFill</Fill>
<Lines>A_GaugeLines</Lines>
<LinesFill>A_GaugeLinesFill</LinesFill>
<EndCapLeft>A_GaugeEndCapLeft</EndCapLeft>
<EndCapRight>A_GaugeEndCapRight</EndCapRight>
</GaugeDrawTemplate>
</Gauge>

<Pieces>Target_EmptyCasting_Gauge</Pieces>
<Pieces>Target_Casting_Gauge</Pieces>

You'll need to adjust the numbers to make it sit where you want it, but this should do it. Just make sure you use the templates named in the celtic ui.
