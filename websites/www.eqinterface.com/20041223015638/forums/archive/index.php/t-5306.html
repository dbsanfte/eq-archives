Open the buff_window.xml file..


<Button item = "BW_Buff0_Button">
<ScreenID>Buff0</ScreenID>
<!--<Font>3</Font>-->
<RelativePosition>true</RelativePosition>
<Location>
<X>3</X>
<Y>3</Y>
</Location>
<Size>
<CX>44</CX> <--------------Change
<CY>44</CY> <--------------Change
</Size>
<Style_VScroll>false</Style_VScroll>
<Style_HScroll>false</Style_HScroll>
<Style_Transparent>false</Style_Transparent>
<!--<TooltipReference/> -->
<Style_Checkbox>false</Style_Checkbox>
<!--<RadioGroup/> -->
<ButtonDrawTemplate>
<Normal>BlueIconBackground</Normal>
<NormalDecal>BuffIcons</NormalDecal>
</ButtonDrawTemplate>
<!--
<SoundPressed/>
<SoundUp/>
<SoundFlyby/>
-->
<DecalOffset>
<X>2</X>
<Y>2</Y>
</DecalOffset>
<DecalSize>
<CX>40</CX> <---------------Change
<CY>40</CY> <---------------Change
</DecalSize>
</Button>


For each buffbutton 0-14
