Find:
<Include>EQUI_LargeDialogWnd.xml</Include>

<Include>EQUI_PvPMerchantWnd.xml</Include>
<Include>EQUI_PvPLeaderBoardWnd.xml</Include>
<Include>EQUI_PvPStatsWnd.xml</Include>
